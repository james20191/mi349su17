 template<typename T>

 template <typename T>
 struct Element{
 public:
     T data_;
     Element* next_;

     Element()=default;
     Element(T d) : data_(d), next_(nullptr) {};
     friend ostream& operator<<(ostream& out, Element e){
         //Prints out data
         out << e.data_;
         return out;
     };
     friend class DJSet<T>;
 };

 template<typename T>
 class DJSet{
     //private:
 private:
     vector<Element<T>* > v_;
     // Element<T>* copy_one_list(Element<T>*);
 public:
     DJSet() = default;
     DJSet(T);
     DJSet(vector<T>);
     DJSet(const DJSet&);
     DJSet& operator=(DJSet);
     ~DJSet();
     Element<T>* member(T);
     Element<T>* add(T);
     Element<T>* join(T, T);

     friend ostream& operator<<(ostream& out, const DJSet<T>& ds){
         //If the size of the vector is 0 just print empty
         if(ds.v_.size() == 0){
             out << "empty";
         }
         //If not empty and print the element out accordigly
         else{
             //Iterate through the vector
             for(size_t i = 0; i < ds.v_.size(); i++){
                 if(i != ds.v_.size() - 1){
                     out << *ds.v_[i] << "->" <<endl;
                 }
                 else{
                     out << *ds.v_[i] << "->";
                 }
             }
         }
         return out;
     };
 };

 //T val conctructor
 template<typename T>
 DJSet<T>::DJSet(T val){
     //Change the the value of val from T to Element<T>*
     Element<T>* n = new Element<T>(val);
     //Add the element to the vector
     v_.push_back(n);
 }

 //Vector conctructor
 template<typename T>
 DJSet<T>::DJSet(vector<T> vec){
     //Iterate through the vector
     for(size_t i = 0; i != vec.size(); i++){
         //Change the the value of vec[i] from T to Element<T>*
         Element<T>* n = new Element<T>(vec[i]);
         //Add the element ot the vector;
         v_.push_back(n);

     }
 }

 //Member function - looks for val in the vector
 template<typename T>
 Element<T>* DJSet<T>::member(T val){
     //Initialize result to nullprt
     Element<T>* result = nullptr;
     //Iterates through the vector to see if val
     //is in there
     for(size_t i = 0; i < v_.size(); i++){
         if(v_[i]->data_ == val){
             result = v_[i];
         }
     }
     return result;
 }

 //Add Function - Add the name if it is not found in the vector
 template<typename T>
 Element<T>* DJSet<T>::add(T name){
     Element<T>* result = nullptr;
     Element<T>* n = new Element<T>(name);
     Element<T>* ptr;
     ptr = member(name);
     //Checks if ptr is null or not, if null then
     //add the name into the vector
     if(ptr == nullptr){
         v_.push_back(n);
         result = n;
     }
     return result;

 }

 //Copy conctructor
 template<typename T>
 DJSet<T>::DJSet(const DJSet& ds){
     //Copy the vector
     v_ = ds.v_;
     //Iterate through the vector to copy each memory
     for(size_t i = 0; i < ds.v_.size(); i++){
         v_[i] = ds.v_[i];
     }
 }

     v_ = ds.v_;
     //Iterate through the vector to copy each memory
